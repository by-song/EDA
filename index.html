# success rate - year
df['Date'] = df['Date'].astype('datetime64')
df['Year'] = df['Date'].dt.year

selected_outcomes = ['True Ocean', 'True RTLS', 'True ASDS']
df['Selected_Outcome'] = df['Outcome'].apply(lambda x: 'Success' if any(outcome in x for outcome in selected_outcomes) else 'Not Success')

success_rate_by_year = df.groupby(['Year', 'Selected_Outcome']).size().unstack(fill_value=0)
success_rate_by_year = success_rate_by_year.div(success_rate_by_year.sum(axis=1), axis=0)

plt.figure(figsize=(10, 6))
success_rate_by_year.plot(kind='area', stacked=True, cmap='viridis', alpha=0.7)
plt.title('Outcome Rate Over Time')
plt.xlabel('Year')
plt.ylabel('Outcome Rate')
plt.legend(bbox_to_anchor=(1.05, 1), loc='upper left')
plt.show()
